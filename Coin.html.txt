<script>
// 计算并更新表25-3
function updateTable() {
    const inputs = document.querySelectorAll('.group-input');
    const mCells = document.querySelectorAll('.m-cell');
    const freqCells = document.querySelectorAll('.freq-cell');

    let cumulative = 0;  // 累计“正面向上”的次数

    inputs.forEach((input, index) => {
        const val = parseInt(input.value, 10);
        // 组内没填或填错就当 0
        const groupCount = isNaN(val) ? 0 : val;
        cumulative += groupCount;   // 累加到当前组

        const n = 50 * (index + 1); // 当前列对应的抛掷总次数

        // 更新 m
        mCells[index].textContent = cumulative > 0 ? cumulative : '';

        // 更新频率 m/n，保留两位小数
        if (cumulative > 0) {
            const freq = cumulative / n;
            freqCells[index].textContent = freq.toFixed(2);
        } else {
            freqCells[index].textContent = '';
        }
    });
}

// 让输入时也自动更新（不点按钮也可以）
document.querySelectorAll('.group-input').forEach(input => {
    input.addEventListener('input', updateTable);
});
</script>
